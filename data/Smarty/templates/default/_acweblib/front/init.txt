//----------------------------------------------------------------------------------------------------
## テストサイトかどうかを定義する
//----------------------------------------------------------------------------------------------------
if(strpos(SITE_ROOT_DIR, '/t/') === false){
	//テストサイトでない場合
	define(ABTEST_FLAG, false);
}
else{
	//テストサイトである場合
	define(ABTEST_FLAG, true);
}



//----------------------------------------------------------------------------------------------------
## <meta property="og:foo">を定義する
//----------------------------------------------------------------------------------------------------
if(!ABTEST_FLAG){
	//テストサイトでない場合
	define(OG_URL_CONTENT, 'https://'.$server_name.$request_uri);
	define(OG_IMAGE_CONTENT, 'https://'.$server_name.SITE_ROOT_DIR.'img/og-image.png');
}
else{
	//テストサイトである場合
	define(OG_URL_CONTENT, str_replace('/t/', '/', 'https://'.$server_name.$request_uri));
	define(OG_IMAGE_CONTENT, str_replace('/t/', '/', 'https://'.$server_name.SITE_ROOT_DIR.'img/og-image.png'));
}



//----------------------------------------------------------------------------------------------------
